        robie: 'Robie',
        name: str,

    ) -> None:
        """
        Initialize instance for class using provided parameters.

            base=self,
            name=name,
            status='initial')

        self.__robie = robie
        self.__name = name
        self.__params = params

        self.__post__()

        robie.logger.log_d(
            base=self,

            status='created')

    def __lt__(
        self,

    ) -> bool:
        """
        Built-in method for comparing this instance with another.
